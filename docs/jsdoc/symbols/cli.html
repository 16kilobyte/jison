<static> cli.main(opts, opts['module-type'], opts['parser-type'])
Generates a parser and writes it to a file. 
 // grammar to process is not json and contains grammars for lexer and
 // parser.
 var jisonCli = require('./node_modules/jison/lib/cli.js');
 var options = {
     file : "myfile.jison",
     moduleName : "myModule"
 };
 jisonCli.main(options);
 // grammar to process is not json and is divided into two files containing
 // the grammars for the lexer and parser seperately.
 var jisonCli = require('./node_modules/jison/lib/cli.js');
 var options = {
     file : "myfile.y",
     lexfile : "myfile.l",
     moduleName : "myModule"
 };
 jisonCli.main(options);
 // grammar to process is in json format, desired module type is amd, and
 // desired parser type is lr0.
 var jisonCli = require('./node_modules/jison/lib/cli.js');
 var options = {
     file : "myfile.json",
     moduleName : "myModule",
     json : true,
     "module-type" : "amd",
     "parser-type" : "lr0"
 };
 jisonCli.main(options);
Parameters:
{Object} opts Optional, Default: {}
Options object.
{String} opts.file Optional, Default: null
Path to a file containing a grammar. If no file is specified input will be read from stdin.
{String} opts.lexfile Optional, Default: null
Path to a file containing a lexical grammar.
{String} opts.outfile Optional, Default: opts.file.replace(/\.[^.]*$/, ".js")
The path and filename where the parser should be written to. Defaults to the path and filename given for file with the file extension replaced by js.
{Boolean} opts.json Optional, Default: false
Set to true if `file` is in json format.
{String} opts.moduleName Optional, Default: "parser"
The internal name for your module in the generated parser.
{Boolean} opts.debug Optional, Default: false
Debug mode.
{String} opts['module-type'] Optional, Default: "commonjs"
The module type of the generated parser. Options are: commonjs, amd, and js.
{String} opts['parser-type'] Optional, Default: "lalr"
The type of parser to generate. Options are: lr0, slr, lalr, and lr.
