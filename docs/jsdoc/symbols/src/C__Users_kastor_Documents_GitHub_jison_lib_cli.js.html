 14 /**

 18 var cli = exports;
 19 /**
 20  * Generates a parser and writes it to a file.
 21  * @param {Object} [opts = {}]
 22  *  Options object.
 23  * @param {String} [opts.file = null]
 24  *  Path to a file containing a grammar. If no file is specified input will be
 25  *  read from stdin.
 26  * @param {String} [opts.lexfile = null]
 27  *  Path to a file containing a lexical grammar.
 28  * @param {String} [opts.outfile = opts.file.replace(/\.[^.]*$/, ".js")]
 29  *  The path and filename where the parser should be written to. Defaults to the
 30  *  path and filename given for <code>file</code> with the file extension
 31  *  replaced by <code>js</code>.
 32  * @param {Boolean} [opts.json = false]
 33  *  Set to true if `file` is in json format.
 34  * @param {String} [opts.moduleName = "parser"]
 35  *  The internal name for your module in the generated parser.
 36  * @param {Boolean} [opts.debug = false] Debug mode.
 37  * @param {String} [opts['module-type'] = "commonjs"]
 38  *  The module type of the generated parser. Options are: <code>commonjs</code>,
 39  *  <code>amd</code>, <code>and</code> <code>js</code>.
 40  * @param {String} [opts['parser-type'] = "lalr"]
 41  *  The type of parser to generate. Options are: <code>lr0</code>,
 42  *  <code>slr</code>, <code>lalr</code>, and <code>lr</code>.
 43  * @example
 44  *  // grammar to process is not json and contains grammars for lexer and
 45  *  // parser.
 46  *  var jisonCli = require('./node_modules/jison/lib/cli.js');
 47  *  var options = {
 48  *      file : "myfile.jison",
 49  *      moduleName : "myModule"
 50  *  };
 51  *  jisonCli.main(options);
 52  * @example
 53  *  // grammar to process is not json and is divided into two files containing
 54  *  // the grammars for the lexer and parser seperately.
 55  *  var jisonCli = require('./node_modules/jison/lib/cli.js');
 56  *  var options = {
 57  *      file : "myfile.y",
 58  *      lexfile : "myfile.l",
 59  *      moduleName : "myModule"
 60  *  };
 61  *  jisonCli.main(options);
 62  * @example
 63  *  // grammar to process is in json format, desired module type is amd, and
 64  *  // desired parser type is lr0.
 65  *  var jisonCli = require('./node_modules/jison/lib/cli.js');
 66  *  var options = {
 67  *      file : "myfile.json",
 68  *      moduleName : "myModule",
 69  *      json : true,
 70  *      "module-type" : "amd",
 71  *      "parser-type" : "lr0"
 72  *  };
 73  *  jisonCli.main(options);
 74  */
 75 cli.main = function cli_main(opts) {
 76     "use strict";
 77     opts = opts || {};
 78     /**
 79      * Generates a parser as a string.
 80      * @param {String} raw Contents of the file named in <code>opts.file</code>
 81      *  argument of {@link cli.main}
 82      * @param {String} lex Contents of the file named in <code>opts.lexfile</code>
 83      *  argument of {@link cli.main}
 84      * @param {Object} Options object supplied to {@link cli.main}
 85      * @private
 86      */
 87     function processGrammar(raw, lex, opts) {
 88         var grammar,
 89         parser;
 90         if (!opts.json) {
 91             grammar = cli.processGrammars(raw, lex, opts.json);
 92         }
 93         parser = cli.generateParserString(opts, grammar);
 94         return parser;
 95     }
 96     /**
 97      * Processes input from a file.
 98      * @private
 99      * @requires <a href="http://nodejs.org/api/fs.html">fs</a>
100      * @requires <a href="http://nodejs.org/api/path.html">path</a>
101      */
102     function processInputFile () {
103         var fs = require('fs');
104         var path = require('path');
105 
106         // getting raw files
107         var lex;
108         if (opts.lexfile) {
109             lex = fs.readFileSync(path.normalize(opts.lexfile), 'utf8');
110         }
111         var raw = fs.readFileSync(path.normalize(opts.file), 'utf8');
112 
113         // making best guess at json mode
114         opts.json = path.extname(opts.file) === '.json' || opts.json;
115 
116         // setting output file name and module name based on input file name
117         // if they aren't specified.
118         var name = path.basename((opts.outfile || opts.file));
119         /*jslint regexp: true */
120         name = name.replace(/\..*$/g, '');
121         /*jslint regexp: false */
122         opts.outfile = opts.outfile || (name + '.js');
123         if (!opts.moduleName && name) {
124             opts.moduleName = name.replace(/-\w/g,
125                     function (match) {
126                     return match.charAt(1).toUpperCase();
127                 });
128         }
129 
130         var parser = processGrammar(raw, lex, opts);
131         fs.writeFileSync(opts.outfile, parser);
132     }
133     /**
134      * Reads from stdin and calls the callback on the data received.
135      * @param {Function} cb The callback function to execute on the received
136      *  data.
137      * @private
138      */
139     function readin(cb) {
140         var stdin = process.openStdin(),
141         data = '';
142 
143         stdin.setEncoding('utf8');
144         stdin.addListener('data', function (chunk) {
145             data += chunk;
146         });
147         stdin.addListener('end', function () {
148             cb(data);
149         });
150     }
151     /**
152      * Processes input from stdin.
153      * @private
154      */
155     function processStdin () {
156         readin(function (raw) {
157             console.log(processGrammar(raw, null, opts));
158         });
159     }
160     
161     // if an input file wasn't given, assume input on stdin
162     if (opts.file) {
163         processInputFile();
164     } else {
165         processStdin();
166     }
167 };
168 /**
169  * Generates a parser and returns it as a string.
170  * @param {Object} [opts = {}]
171  *  An options object.
172  * @param {Boolean} [opts.json]
173  *  See the description in {@link cli.main}
174  * @param {String} [opts.moduleName]
175  *  See the description in {@link cli.main}
176  * @param {Object} [opts.debug]
177  *  See the description in {@link cli.main}
178  * @param {Object} [opts['parser-type']]
179  *  See the description in {@link cli.main}
180  * @param {Object} [opts['module-type']]
181  *  See the description in {@link cli.main}
182  * @param {String|Object} grammar The grammar to generate a parser from.
183  * @returns {String} Returns the generated parser as a string.
184  * @requires <a href="https://npmjs.org/package/jison/">jison</a>
185  */
186 cli.generateParserString = function generateParserString(opts, grammar) {
187     "use strict";
188     opts = opts || {};
189     var jison = require('./jison.js');
190 
191     var settings = grammar.options || {};
192 
193     if (opts['parser-type']) {
194         settings.type = opts['parser-type'];
195     }
196     settings.debug = opts.debug;
197     if (!settings.moduleType) {
198         settings.moduleType = opts['module-type'];
199     }
200 
201     var generator = new jison.Generator(grammar, settings);
202     return generator.generate(settings);
203 };
204 /**
205  * Processes grammar files of various format.
206  * @param {String} file Contents of a jison grammar file.
207  * @param {String} [lexFile] Contents of a lexer grammar file.
208  * @param {Boolean} [jsonMode = false] Set to true if <code>file</code> is in
209  *  json format.
210  * @returns {Object} Returns the parsed grammar object.
211  * @requires <a href="https://npmjs.org/package/ebnf-parser">ebnf-parser</a>
212  * @requires <a href="https://npmjs.org/package/cjson">cjson</a>
213  * @requires <a href="https://npmjs.org/package/lex-parser">lex-parser</a>
214  */
215 cli.processGrammars = function processGrammars(file, lexFile, jsonMode) {
216     "use strict";
217     lexFile = lexFile || false;
218     jsonMode = jsonMode || false;
219     var ebnfParser = require('ebnf-parser');
220     var cjson = require('cjson');
221     var grammar;
222     try {
223         if (jsonMode) {
224             grammar = cjson.parse(file);
225         } else {
226             grammar = ebnfParser.parse(file);
227         }
228     } catch (e) {
229         throw new Error('Could not parse jison grammar');
230     }
231     try {
232         if (lexFile) {
233             grammar.lex = require('lex-parser').parse(lexFile);
234         }
235     } catch (e) {
236         throw new Error('Could not parse lex grammar');
237     }
238     return grammar;
239 };
240 /**
241  * Initialization function, grabs commandline arguments and passes them to
242  *  {@link cli.main} if this script was called from the commandline.
243  * @private
244  * @methodOf cli
245  * @requires <a href="https://npmjs.org/package/nomnom">nomnom</a>
246  */
247 function cli_init () {
248     "use strict";
249     /**
250      * Gets options from the commandline.
251      * @private
252      * @requires <a href="https://npmjs.org/package/nomnom">nomnom</a>
253      */
254     function getCommandlineOptions () {
255         var version = require('../package.json').version;
256         var opts = require("nomnom")
257             .script('jison')
258             .option('file', {
259                 flag : true,
260                 position : 0,
261                 help : 'file containing a grammar'
262             })
263             .option('lexfile', {
264                 flag : true,
265                 position : 1,
266                 help : 'file containing a lexical grammar'
267             })
268             .option('json', {
269                 abbr : 'j',
270                 flag : true,
271                 help : 'force jison to expect a grammar in JSON format'
272             })
273             .option('outfile', {
274                 abbr : 'o',
275                 metavar : 'FILE',
276                 help : 'Filename and base module name of the generated parser'
277             })
278             .option('debug', {
279                 abbr : 't',
280                 flag : true,
281             default:
282                 false,
283                 help : 'Debug mode'
284             })
285             .option('module-type', {
286                 abbr : 'm',
287             default:
288                 'commonjs',
289                 metavar : 'TYPE',
290                 help : 'The type of module to generate (commonjs, amd, js)'
291             })
292             .option('parser-type', {
293                 abbr : 'p',
294             default:
295                 'lalr',
296                 metavar : 'TYPE',
297                 help : 'The type of algorithm to use for the parser (lr0, slr,' +
298                     'lalr, lr)'
299             })
300             .option('version', {
301                 abbr : 'V',
302                 flag : true,
303                 help : 'print version and exit',
304                 callback : function () {
305                     return version;
306                 }
307             }).parse();
308             
309         return opts;
310     } // end of getCommandlineOptions
311 
312     if (require.main === module) {
313         var opts = getCommandlineOptions();
314         cli.main(opts);
315     }
316 }
317
